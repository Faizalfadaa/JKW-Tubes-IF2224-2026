program NormalBangetLengkapProgramnya;

type
  AngkaArray = array[5] of integer;

  DataLoop = record
    i: integer;
    j: integer;
    status: string;
  end;

var
  data: DataLoop;
  arr: AngkaArray;
  hasil: integer;

procedure TulisPesan(teks: string);
begin
  writeln(teks);
end;

function Hitung(x: integer): integer;
begin
  Hitung := (x div 3) MOD 2;
end;

begin
  data.i := 0;
  data.j := 0;
  data.status := 'awal';

  arr[1] := 10;
  arr[2] := 20;

  while data.i <= 10 do
    data.i := data.i + 1;

  for data.j := 10 downto 1 do
    data.j := data.j - 1;

  for data.i := 1 to 3 do
    data.i := Hitung(data.i);

  repeat
    data.i := data.i + 1;
  until data.i >= 10;

  if data.i >= 10 then
    data.status := 'selesai'
  else
    data.status := 'belum';

  hasil := data.i;

  case hasil of
  0:
    writeln('nol');
  1:
    writeln('satu');
  2:
    writeln('dua');
  end;

  TulisPesan(data.status);
end.
