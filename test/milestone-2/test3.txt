program TestConstEnumRange;

const
  MAX == 100;
  MIN == -10;
  PI == 3.14;
  MSG == 'hello';
  CH == 'a';

type
  Direction == (north, south, east, west);
  Month == (jan, feb, mar, apr, may, jun);
  Letter == 'a'..'z';
  UpperLetter == 'A'..'Z';

var
  d: integer;
  i: integer;

begin
  d := 5;
  i := MAX;

  case d of
    0: writeln('nol');
    1, 2, 3: writeln('kecil');
    4, 5: writeln('sedang');
    6, 7, 8, 9: writeln('besar')
  end;

  case i of
    100: writeln('maksimum')
  end;
end.
