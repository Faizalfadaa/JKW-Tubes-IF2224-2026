function Abs(n: integer): integer;
begin
  if n < 0 then
    Abs := -n
  else
    Abs := n;
end;

function Clamp(val, lo, hi: integer): integer;
begin
  if val < lo then
    Clamp := lo
  else if val > hi then
    Clamp := hi
  else
    Clamp := val;
end;
